现代JavaScript ES6+特性详解

2023年9月25日 阅读时间: 12分钟 JavaScript

ES6 (ECMAScript 2015) 及其后续版本引入了许多强大特性，彻底改变了JavaScript开发方式。本文将系统梳理这些现代特性及其实际应用。

变量声明：let和const

ES6引入了let和const，解决了var的块级作用域问题：

// var的问题
for (var i = 0; i < 5; i++) {
  setTimeout(() => console.log(i), 100); // 5 5 5 5 5
}

// let的块级作用域
for (let i = 0; i < 5; i++) {
  setTimeout(() => console.log(i), 100); // 0 1 2 3 4
}

const PI = 3.14159;
// PI = 3.14; // TypeError: Assignment to constant variable.

箭头函数

箭头函数提供了更简洁的语法和不同的this绑定：

// 传统函数
const add = function(a, b) {
  return a + b;
};

// 箭头函数
const add = (a, b) => a + b;

// this绑定不同
const obj = {
  name: 'MASHIJIAN',
  sayHi: function() {
    setTimeout(function() {
      console.log(this.name); // undefined
    }, 100);
    
    setTimeout(() => {
      console.log(this.name); // MASHIJIAN
    }, 100);
  }
};

模板字符串

支持多行字符串和变量插值：

const name = 'MASHIJIAN';
const age = 30;

// 传统拼接
const greeting1 = 'Hello, ' + name + '! You are ' + age + ' years old.';

// 模板字符串
const greeting2 = `Hello, ${name}!
You are ${age} years old.`;

const html = `
  

    
Welcome ${name}

    
Age: ${age}

  

`;

解构赋值

从数组或对象中提取值并赋值：

// 数组解构
const [first, second, ...rest] = [1, 2, 3, 4, 5];
console.log(first); // 1
console.log(rest); // [3, 4, 5]

// 对象解构
const {name, age, city = 'Beijing'} = {
  name: 'MASHIJIAN',
  age: 30
};
console.log(city); // Beijing

// 函数参数解构
function greet({name, greeting = 'Hi'}) {
  console.log(`${greeting}, ${name}!`);
}
greet({name: 'MASHIJIAN'});

扩展运算符(...)

用于数组展开、对象合并等：

// 数组展开
const arr1 = [1, 2, 3];
const arr2 = [...arr1, 4, 5]; // [1, 2, 3, 4, 5]

// 函数参数
function sum(a, b, c) {
  return a + b + c;
}
const numbers = [1, 2, 3];
console.log(sum(...numbers)); // 6

// 对象合并
const obj1 = {name: 'MASHIJIAN'};
const obj2 = {age: 30};
const merged = {...obj1, ...obj2, city: 'Beijing'};
console.log(merged); // {name: 'MASHIJIAN', age: 30, city: 'Beijing'}

默认参数和剩余参数

function makeRequest(url, options = {}) {
  const {method = 'GET', headers = {}} = options;
  // 发送请求逻辑
}

function log(...messages) {
  console.log.apply(console, messages);
}
log('Debug:', 'This is a message', {data: 'value'});

模块系统

ES6原生支持模块导入导出：

// math.js
export const PI = 3.14159;
export function circleArea(r) {
  return PI * r * r;
}
export default function squareArea(s) {
  return s * s;
}

// main.js
import squareArea, {PI, circleArea} from './math.js';
console.log(squareArea(5)); // 25
console.log(circleArea(3)); // 28.274

类和继承

class Animal {
  constructor(name) {
    this.name = name;
  }
  
  speak() {
    console.log(`${this.name} makes a sound.`);
  }
}

class Dog extends Animal {
  constructor(name, breed) {
    super(name);
    this.breed = breed;
  }
  
  speak() {
    super.speak();
    console.log(`${this.name} barks.`);
  }
}

const dog = new Dog('Buddy', 'Golden Retriever');
dog.speak();

Promise和async/await

// Promise
function fetchData() {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      if (Math.random() > 0.5) {
        resolve({data: 'Success'});
      } else {
        reject(new Error('Failed'));
      }
    }, 1000);
  });
}

fetchData()
  .then(result => console.log(result.data))
  .catch(error => console.error(error.message));

// async/await
async function getData() {
  try {
    const result = await fetchData();
    console.log(result.data);
  } catch (error) {
    console.error(error.message);
  }
}

总结

ES6+特性极大地提升了JavaScript的表达力和开发效率。熟练掌握这些特性是现代前端开发者的必备技能。通过实践这些新语法，开发体验将显著改善。
